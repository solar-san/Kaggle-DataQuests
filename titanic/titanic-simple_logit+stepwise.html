data_loader <- function(filename, dataset, path = "data/", kaggle = FALSE, kaggle_path = NULL) {
  
  if(kaggle == FALSE) {
  path <- paste0(path, filename)
  read.csv(
    unz(
      path,
      dataset
    )
  )
  } else {
    read.csv(
      kaggle_path
    )
  }
  
}

factorize_data <- function(dataframe) {
  dataframe %>% 
  select(-Name, -Ticket) %>% 
  mutate(
    Pclass = as.factor(Pclass),
    Sex = as.factor(Sex),
    # SibSp = as.factor(SibSp),
    # Parch = as.factor(Parch),
    Cabin = as.factor(Cabin),
    Embarked = as.factor(Embarked)
  )
}

Variable de-scaler

Code 
de_scale <- function(x) {
  x * attr(x, 'scaled:scale') + attr(x, 'scaled:center')
}
